#include "bits/stdc++.h"
using namespace std;
const int MAXN=1000+5,MAXM=10000+5;
int v[MAXN],c[MAXM],f[MAXN][MAXM];
int main()
{
    int n=0,m=0,k=0;
    scanf("%d%d%d",&n,&m,&k);
    for(int i=1; i<=n; i++)
    {
        scanf("%d%d",&v[i],&c[i]);
    }
    memset(f,0,sizeof(m));
    for(int i=1; i<=n; i++)
    {
        for(int j=1; j<=m; j++)
        {
            if(j<v[i])
                f[i][j]=0;
            else
            {
                f[i][j]=max(f[i-1][j-v[i]]+c[i],f[i-1][j]);
            }

        }
    }
    printf("%d\n",f[n][m]+k);
    return 0;
}
